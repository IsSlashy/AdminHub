  generateContractPdf() {
    console.log('le job', this.job);
    this.apollo
      .mutate({
        mutation: GENERATE_CONTRACT,
        variables: {
          offerId: this.job.finalOffer.id,
        },
        refetchQueries: [{ query: BILLING }],
      })
      .subscribe(
        ({ data }: any) => {
          this.modalConfirmed.openModal();
        },
        (error) => {
          console.log('error a la génération du contrat', error);
        }
      );
  }
